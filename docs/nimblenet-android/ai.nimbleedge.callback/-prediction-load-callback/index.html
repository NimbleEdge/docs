interface PredictionLoadCallback
Content copied to clipboard

This interface is overridden to implement callbacks for asynchronous prediction from the models.

Example Code

object : PredictionLoadCallback {
      override fun onSuccess(output: ModelOutput) {
          Log.d(NE_LOG_TAG, "predictAsync:onSuccess=>" + Thread.currentThread().name)
          val text = output["probabilities"]?.data?.joinToString(",") {
              String.format("%.2f",it)
          }
          val text2 = output["label"]?.data?.joinToString(",") {
              String.format("%d",it)
          }
      }

      override fun onFailure(failure: NimbleNetResult.Failure) {
          Log.d(NE_LOG_TAG, "predictAsync:onFailure" + Thread.currentThread().name)
      }

      override fun onFallBack(fallBack: NimbleNetResult.FallBack) {
          Log.d(NE_LOG_TAG, "predictAsync:onFailure" + Thread.currentThread().name)
      }
  }
