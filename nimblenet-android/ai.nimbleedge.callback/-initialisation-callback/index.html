interface InitialisationCallback
Content copied to clipboard

This interface is overridden to implement callbacks for asynchronous initialization of the ai.nimbleedge.NimbleNet

Example Code

object : InitialisationCallback {
     override fun onSuccess() {
         Log.d(NE_LOG_TAG, "initAsync:onSuccess")
     }

     override fun onFailure(failure: NimbleNetResult.Failure) {
         Log.d(NE_LOG_TAG, "initAsync:onFailure=>$failure")
     }

     override fun onFallBack(fallback: NimbleNetResult.FallBack) {
         Log.d(NE_LOG_TAG, "initAsync:onFallBack=>$fallback")
     }
 })
}
